import pandas as pd  # data analysis
import matplotlib.pyplot as plt  # plotting
import s3fs  # S3 access

# Define S3 bucket and table folders
bucket = 'customer-dc-grai-matter-prod'

tables = {
    'dim_patient': 'grai_matter_dim_patient',
    'fct_encounters': 'grai_matter_fct_encounters',
    'fct_medication': 'grai_matter_fct_medication',
    'dim_med_admin': 'grai_matter_dim_med_admin'
}

# Initialize S3 file system
fs = s3fs.S3FileSystem()

# Store summary results
table_shapes = {}
table_columns = {}
unique_patients = {}
date_ranges = {}

# Function to read and concatenate all Parquet files in an S3 folder
def read_full_table(bucket, folder):
    files = fs.ls(f'{bucket}/{folder}')
    file_paths = [f's3://{file}' for file in files if file.endswith('.parquet')]
    df_parts = [pd.read_parquet(fp, filesystem=fs) for fp in file_paths]
    return pd.concat(df_parts, ignore_index=True)

# Analyze each table
for name, folder in tables.items():
    print(f"Reading table: {name}")
    df = read_full_table(bucket, folder)

    table_shapes[name] = df.shape
    table_columns[name] = df.columns.tolist()

    # Count unique patients if column exists
    pat_col = [col for col in df.columns if 'PATIENT_IDENTIFIER' in col.upper()]
    if pat_col:
        unique_patients[name] = df[pat_col[0]].nunique()

    # Compute date ranges
    date_cols = [col for col in df.columns if 'date' in col.lower()]
    for col in date_cols:
        try:
            df[col] = pd.to_datetime(df[col], errors='coerce')
            date_ranges[f'{name}:{col}'] = (df[col].min(), df[col].max())
        except:
            continue

    # Plot histogram of birth years if in patient table
    if name == 'dim_patient' and 'PATIENT_BIRTH_YEAR' in df.columns:
        df['PATIENT_BIRTH_YEAR'].dropna().astype(int).hist(bins=30)
        plt.title("Patient Birth Year Distribution")
        plt.xlabel("Birth Year")
        plt.ylabel("Patient Count")
        plt.grid(True)
        plt.tight_layout()
        plt.show()

# Print summary
print("\nSummary:")
for name in tables:
    print(f"\nTable: {name}")
    print(f"Shape: {table_shapes.get(name)}")
    print(f"Columns: {table_columns.get(name)}")
    print(f"Unique Patients: {unique_patients.get(name, 'N/A')}")
    print("Date Ranges:")
    for k, rng in date_ranges.items():
        if k.startswith(name):
            print(f"  {k.split(':')[1]}: {rng[0]} to {rng[1]}")

Key Join Fields¶

  • DH_ENCOUNTER_ID: joins encounters with medication and admin tables
  • DH_ORDER_ID: connects medication orders with actual administration

Table Functions¶

  • dim_patient: Patient demographic and death status
  • fct_encounters: Each row is a clinical encounter
  • fct_medication: Orders placed during an encounter
  • dim_med_admin: Execution of the medication (dose, time, route)

  • Each encounter can result in multiple medication orders
  • Each order can lead to multiple administration records
