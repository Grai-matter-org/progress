import pandas as pd
import pyarrow.dataset as ds

pd.set_option("display.max_rows", None)
pd.set_option("display.max_columns", None)
pd.set_option("display.width", None)
pd.set_option("display.max_colwidth", None)

BUCKET = "customer-dc-grai-matter-prod"
PID_COL = "PRIMARY_PATIENT_IDENTIFIER"

ENC_START, ENC_END = "ENCOUNTER_START_DT", "ENCOUNTER_END_DT"
MED_START, MED_END = "ORDER_DT", "ORDER_END_DT"

TABLES = [
    {"name": "DIM_PATIENT",    "prefix": "grai_matter_dim_patient",    "date_cols": []},
    {"name": "FCT_MEDICATION", "prefix": "grai_matter_fct_medication", "date_cols": [MED_START, MED_END]},
    {"name": "FCT_ENCOUNTERS", "prefix": "grai_matter_fct_encounters", "date_cols": [ENC_START, ENC_END]},
    {"name": "DIM_MED_ADMIN",  "prefix": "grai_matter_dim_med_admin",  "date_cols": []},
]

def s3path(prefix: str) -> str:
    """Build S3 dataset path with trailing slash."""
    return f"s3://{BUCKET}/{prefix}/"

def get_schema_and_nrows(path: str):
    """Get ALL column names and fast row count via PyArrow Dataset (no full load)."""
    dataset = ds.dataset(path, format="parquet")
    cols = dataset.schema.names
    nrows = dataset.count_rows()
    return cols, nrows

for t in TABLES:
    name = t["name"]
    path = s3path(t["prefix"])
    date_cols = t["date_cols"]

    # --- Shape & ALL columns ---
    cols, n_rows = get_schema_and_nrows(path)
    n_cols = len(cols)
    print(f"\n================ {name} ================")
    print(f"Shape: ({n_rows}, {n_cols})")

    # Print ALL column names with index for readability
    col_df = pd.DataFrame({"position": range(1, n_cols + 1), "column_name": cols})
    print("\nAll columns:")
    display(col_df)  
    
    # --- Number of unique patients ---
    pid_ser = pd.read_parquet(path, engine="pyarrow",
                              columns=[PID_COL], dtype_backend="pyarrow")[PID_COL]
    n_unique = pid_ser.nunique(dropna=True)
    print(f"\nNumber of unique patients ({PID_COL}): {n_unique}")

    # --- Date ranges (only for specified tables) ---
    if date_cols:
        df_dates = pd.read_parquet(path, engine="pyarrow",
                                   columns=date_cols, dtype_backend="pyarrow")
        print("\nDate ranges:")
        for dc in date_cols:
            s = pd.to_datetime(df_dates[dc], errors="coerce", utc=True)
            print(f"  {dc}: min={s.min()}   max={s.max()}")

# ===== Patient Age Histogram from DIM_PATIENT (using PATIENT_BIRTH_YEAR) =====
import matplotlib.pyplot as plt

dim_patient_prefix = next(t["prefix"] for t in TABLES if t["name"] == "DIM_PATIENT")
path = s3path(dim_patient_prefix)

YEAR_COL = "PATIENT_BIRTH_YEAR"   
try:
    df_pat = pd.read_parquet(path, engine="pyarrow",
                             columns=[PID_COL, YEAR_COL],
                             dtype_backend="pyarrow")
except TypeError:
    df_pat = pd.read_parquet(path, engine="pyarrow",
                             columns=[PID_COL, YEAR_COL])

# to numeric year; invalid -> NaN
year = pd.to_numeric(df_pat[YEAR_COL], errors="coerce")
current_year = pd.Timestamp("now", tz="UTC").year

# compute age (years); drop NaN only 
age_years = current_year - year
age_clean = age_years.dropna()

# plot histogram
plt.figure()
plt.hist(age_clean, bins=40)
plt.title("Patient Age Distribution (DIM_PATIENT)")
plt.xlabel("Age (years)")
plt.ylabel("Count")
plt.show()

Task 4: Schema Mapping and Linkage

Brief Function of Each Table

DIM_PATIENT: Patient index and demographics

FCT_ENCOUNTERS: Encounter (visit) record

DIM_ADT: Patient movement events (admit, transfer, discharge) within a patient encounter

DIM_ENC_EVENT_RECORDS: Workflow timestamp events during the encounter

FCT_MEDICATION: Medication orders/prescriptions

DIM_MED_ADMIN: Medication administrations

FCT_ICD_CODES: Diagnosis codes

FCT_PROCEDURE_CODES: Procedure/Operation codes

Table Relationships Key Joins

Patient Encounters: by PRIMARY_PATIENT_IDENTIFIER (1:N; one patient can have many encounters)

Encounters ADT (bed/location moves): by DH_ENCOUNTER_ID (1:N)

Encounters Event Records (workflow events): by DH_ENCOUNTER_ID (1:N)

Encounters Medication Orders: by DH_ENCOUNTER_ID (1:N); each order has a DH_ORDER_ID

Medication Orders Medication Administrations: by DH_ORDER_ID (1:N; one order can have many administrations)

Encounters ICD (diagnoses): by DH_ENCOUNTER_ID (1:N)

Encounters Procedures (operations): by DH_ENCOUNTER_ID (1:N)
